1   package util.javafx;

3   import javafx.stage.FileChooser;
4   import javafx.stage.Stage;
5   
6   import java.io.File;
7   import java.util.Optional;
8   
9   /**
10   * Class that provides option for browse after files on the computer that we would like to read in. This class can accept only json file format.
11   */
12  
13  public final class FileChooserHelper {
14  
15      /**
16       * show method provides option to browse file on the computer.
17       * @param isOpen tells if the page is open.
18       * @param stage stage we want to add to.
19       * @return returns the dialog page that lets us read in files.
20       */
21      public static Optional<File> show(boolean isOpen, Stage stage) {
22          FileChooser fileChooser = new FileChooser();
23          fileChooser.getExtensionFilters().add(new FileChooser.ExtensionFilter("json documents", "*.json"));
24          if (isOpen) {
25              fileChooser.setTitle("Browse a JSON file to load the scoreboard");
26              return Optional.ofNullable(fileChooser.showOpenDialog(stage));
27          } else {
28              fileChooser.setTitle("Browse a JSON file to save the scoreboard");
29              return Optional.ofNullable(fileChooser.showSaveDialog(stage));
30          }
31      }
32  }
