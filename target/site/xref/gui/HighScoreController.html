21  import javax.inject.Inject;
22  import java.io.IOException;
23  import java.time.ZonedDateTime;
24  import java.util.List;
25  
26  public final class HighScoreController {
27  
28      @Inject
29      private FXMLLoader fxmlLoader;
30  
31  
32      @FXML
33      private TableView<GameResult> highScoreTable;
34  
35      @FXML
36      private TableColumn<GameResult, String> foxNickName;
37      @FXML
38      private TableColumn<GameResult, String> dogsNickName;
39  
40      @FXML
41      private TableColumn<GameResult, String> winner;
42  
43      @FXML
44      private TableColumn<GameResult, Integer> steps;
45  
46  
47      @FXML
48      private TableColumn<GameResult, ZonedDateTime> created;
49  
50      private static final Logger logger = LogManager.getLogger(HighScoreController.class);
51  
52      @FXML
53      private void initialize() {
54          logger.debug("Loading high scores...");
55  
56          List<GameResult> highScoreList = ResultRepository.getGameResults();
57          logger.info(highScoreList);
58  
59          foxNickName.setCellValueFactory(new PropertyValueFactory<>("player1"));
60          dogsNickName.setCellValueFactory(new PropertyValueFactory<>("player2"));
61          steps.setCellValueFactory(new PropertyValueFactory<>("steps"));
62          winner.setCellValueFactory(new PropertyValueFactory<>("winner"));
63          created.setCellValueFactory(new PropertyValueFactory<>("created"));
64  
65          ObservableList<GameResult> observableResult = FXCollections.observableArrayList();
66          observableResult.addAll(highScoreList);
67  
68          highScoreTable.setItems(observableResult);
69      }
70  
71      public void handleRestartButton(ActionEvent actionEvent) throws IOException {
72          Stage stage = (Stage) ((Node) actionEvent.getSource()).getScene().getWindow();
73          logger.debug("{} is pressed", ((Button) actionEvent.getSource()).getText());
74          ControllerHelper.loadAndShowFXML("/fxml/gameOpening.fxml", stage);
75      }
76  
77      public void handleSaveScoreboard(ActionEvent actionEvent) {
78          FileChooserHelper.show(
79                          false,
80                          (Stage) ((Node) actionEvent.getSource()).getScene().getWindow()
81                  )
82                  .ifPresent(file -> {
83                      try {
84                          ResultRepository.writeResult(file);
85                      } catch (IOException e) {
86                          throw new RuntimeException(e);
87                      }
88                  });
89  
90          logger.debug("{} is pressed", ((Button) actionEvent.getSource()).getText());
91      }
92  }
