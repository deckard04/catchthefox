40      public void loadScoreBoard(ActionEvent actionEvent) throws IOException{
41          FileChooserHelper.show(
42                      true,
43                      (Stage) ((Node) actionEvent.getSource()).getScene().getWindow()
44              )
45              .ifPresent(file -> {
46                  try {
47                      ResultRepository.loadFromFile(file);
48                  } catch (IOException ex) {
49                      logger.error(ex.getMessage());
50                  }
51              });
52      }
53  
54      public void startGame(ActionEvent actionEvent) throws IOException {
55          if (foxNickName.getText().isEmpty() || dogsNickName.getText().isEmpty() || foxNickName.getText().length() > 14 || dogsNickName.getText().length() > 14 || foxNickName.getText().equals(dogsNickName.getText())) {
56              nickNameErrorLabel.setText("Hiba! A nevek nem lehetnek üresek, azonosak, vagy 14 karakternél hosszabbak!");
57          } else {
58              Stage stage = (Stage) ((Node) actionEvent.getSource()).getScene().getWindow();
59  
60              FXMLLoader fxmlLoader = new FXMLLoader();
61              fxmlLoader.setLocation(getClass().getResource("/fxml/gameUI.fxml"));
62              Parent root = fxmlLoader.load();
63              fxmlLoader.<GameController>getController().setFoxNickName(foxNickName.getText());
64              fxmlLoader.<GameController>getController().setDogsNickName(dogsNickName.getText());
65              stage.setResizable(false);
66              stage.setScene(new Scene(root));
67              stage.show();
68              logger.info("The usernames are set to {} (fox) and {} (dogs), loading game scene", foxNickName.getText(), dogsNickName.getText());
69          }
70      }
